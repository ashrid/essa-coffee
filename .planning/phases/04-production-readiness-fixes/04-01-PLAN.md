       ```typescript
       export async function generateMagicToken(email: string): Promise<string> {
         const token = crypto.randomUUID().replace(/-/g, ''); // 32-char hex string
         const expires = new Date(Date.now() + 15 * 60 * 1000); // 15 minutes

         await prisma.verificationToken.create({
           data: {
             identifier: email.toLowerCase(),
             token,
             expires,
           },
         });

         return token;
       }
       ```

    4. Rewrite `verifyMagicToken` to use Prisma with immediate deletion:
       ```typescript
       export async function verifyMagicToken(token: string): Promise<string | null> {
         const record = await prisma.verificationToken.findUnique({
           where: { token },
         });

         if (!record) return null;

         // Delete immediately (one-time use)
         await prisma.verificationToken.delete({
           where: { token },
         });

         // Check expiration after retrieval
         if (record.expires < new Date()) {
           return null;
         }

         return record.identifier;
       }
       ```

    5. Update `sendMagicLink` to await the async `generateMagicToken`:
       - Change `const token = generateMagicToken(email);` to `const token = await generateMagicToken(email);`

    6. Update the authorize callback in NextAuth config to await `verifyMagicToken`:
       - Change `const email = verifyMagicToken(credentials.token as string);` to `const email = await verifyMagicToken(credentials.token as string);`

    NOTE: The function signatures change from sync to async. All call sites must be updated to use await.
