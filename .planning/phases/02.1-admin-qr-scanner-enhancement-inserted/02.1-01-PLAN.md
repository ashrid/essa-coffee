    ```typescript
    /**
     * Extract token from QR code scanned content
     * Supports both URL format (?token=xxx) and raw tokens
     * NEVER logs the scanned token for security
     */
    export function extractTokenFromQR(scannedText: string): { token: string | null; error?: string } {
      // Security: sanitize input - never log scanned content
      const sanitized = scannedText?.trim();

      if (!sanitized) {
        return { token: null, error: 'Empty QR code content' };
      }

      try {
        // Check if it's a URL with token param
        if (sanitized.includes('?token=') || sanitized.startsWith('http')) {
          const url = new URL(sanitized);
          const token = url.searchParams.get('token');
          if (token) {
            return { token: token.trim() };
          }
          return { token: null, error: 'Invalid QR code. No token found in URL.' };
        }

        // Raw token format
        return { token: sanitized };
      } catch {
        // If URL parsing fails, treat as raw token
        return { token: sanitized };
      }
    }

    /**
     * Check if running in secure context (HTTPS or localhost)
     * Required for camera access
     */
    export function isSecureContext(): boolean {
      if (typeof window === 'undefined') return false;
      return window.isSecureContext === true;
    }

    /**
     * Get browser-specific camera permission instructions
     */
    export function getPermissionInstructions(): string {
      if (typeof window === 'undefined') return '';

      const userAgent = navigator.userAgent.toLowerCase();

      if (userAgent.includes('chrome') && !userAgent.includes('edg')) {
        return 'To enable camera: Tap the lock icon in the address bar → Site settings → Camera → Allow.';
      }
      if (userAgent.includes('safari') && !userAgent.includes('chrome')) {
        return 'To enable camera: Open Settings app → Safari → Camera → Allow.';
      }
      if (userAgent.includes('firefox')) {
        return 'To enable camera: Tap the permissions icon in the address bar → Camera → Allow.';
      }
      if (userAgent.includes('samsung')) {
        return 'To enable camera: Tap the menu button → Settings → Site permissions → Camera → Allow.';
      }

      return 'Please enable camera access in your browser settings.';
    }

    /**
     * Check if device is running iOS < 14.3 (no camera support in WKWebView)
     */
    export function isUnsupportedIOS(): boolean {
      if (typeof window === 'undefined') return false;

      const userAgent = navigator.userAgent;
      const match = userAgent.match(/OS (\d+)_(\d+)_?(\d+)?/);

      if (match) {
        const major = parseInt(match[1], 10);
        const minor = parseInt(match[2], 10);
        // iOS 14.3+ required for camera in WKWebView
        if (major < 14 || (major === 14 && minor < 3)) {
          return true;
        }
      }

      return false;
    }
    ```

    ```typescript
    import { describe, it, expect } from 'vitest';
    import { extractTokenFromQR } from './qr-scanner';

    describe('extractTokenFromQR', () => {
      // Valid cases
      it('extracts token from URL with query param', () => {
        const result = extractTokenFromQR('https://example.com/admin/scan?token=abc123');
        expect(result.token).toBe('abc123');
        expect(result.error).toBeUndefined();
      });

      it('extracts token from URL with multiple params', () => {
        const result = extractTokenFromQR('https://example.com/admin/scan?foo=bar&token=xyz789&baz=qux');
        expect(result.token).toBe('xyz789');
        expect(result.error).toBeUndefined();
      });

      it('handles raw token format', () => {
        const result = extractTokenFromQR('raw-token-123');
        expect(result.token).toBe('raw-token-123');
        expect(result.error).toBeUndefined();
      });

      it('trims whitespace from raw tokens', () => {
        const result = extractTokenFromQR('  token-with-spaces  ');
        expect(result.token).toBe('token-with-spaces');
      });

      // Error cases
      it('returns error for empty string', () => {
        const result = extractTokenFromQR('');
        expect(result.token).toBeNull();
        expect(result.error).toBe('Empty QR code content');
      });

      it('returns error for whitespace-only string', () => {
        const result = extractTokenFromQR('   ');
        expect(result.token).toBeNull();
        expect(result.error).toBe('Empty QR code content');
      });

      it('returns error for null/undefined', () => {
        const result = extractTokenFromQR(null as unknown as string);
        expect(result.token).toBeNull();
        expect(result.error).toBe('Empty QR code content');
      });

      it('returns error for URL without token param', () => {
        const result = extractTokenFromQR('https://example.com/admin/scan');
        expect(result.token).toBeNull();
        expect(result.error).toBe('Invalid QR code. No token found in URL.');
      });

      it('returns error for URL with empty token', () => {
        const result = extractTokenFromQR('https://example.com/admin/scan?token=');
        expect(result.token).toBeNull();
        expect(result.error).toBe('Invalid QR code. No token found in URL.');
      });

      // Edge cases
      it('handles URL with special characters in token', () => {
        const result = extractTokenFromQR('https://example.com/admin/scan?token=abc%2B123');
        expect(result.token).toBe('abc+123'); // URL decoded
      });

      it('handles token with query-like content', () => {
        const result = extractTokenFromQR('token?with=special');
        expect(result.token).toBe('token?with=special');
      });
    });
    ```
