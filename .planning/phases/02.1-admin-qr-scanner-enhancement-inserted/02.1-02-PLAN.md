    'use client';

    import { useEffect, useRef, useState, useCallback } from 'react';
    import { Html5QrcodeScanner } from 'html5-qrcode';
    import { AlertCircle, CameraOff, Loader2 } from 'lucide-react';
    import { Button } from '@/components/ui/button';
    import {
      extractTokenFromQR,
      isSecureContext,
      getPermissionInstructions,
      isUnsupportedIOS
    } from '@/lib/qr-scanner';

    interface QRScannerProps {
      onScan: (token: string) => void;
      onError?: (error: string) => void;
    }

    type ScannerState = 'initializing' | 'scanning' | 'error' | 'permission-denied' | 'unsupported';

    export function QRScanner({ onScan, onError }: QRScannerProps) {
      const scannerRef = useRef<Html5QrcodeScanner | null>(null);
      const scanningLockRef = useRef(false); // Prevents duplicate scans
      const initRef = useRef(false); // Tracks if initialization started
      const [state, setState] = useState<ScannerState>('initializing');
      const [errorMessage, setErrorMessage] = useState<string>('');
      const elementId = 'admin-qr-scanner';

      // Calculate responsive qrbox dimensions
      const getQrBoxDimensions = useCallback(() => {
        if (typeof window === 'undefined') return { width: 250, height: 250 };
        const minSize = 200;
        const maxSize = 300;
        const preferredSize = Math.min(window.innerWidth * 0.7, maxSize);
        const size = Math.max(minSize, preferredSize);
        return { width: size, height: size };
      }, []);

      const handleScan = useCallback((decodedText: string) => {
        // SECURITY: Never log decodedText - prevents token leakage

        // Debounce: prevent duplicate scans while processing
        if (scanningLockRef.current) {
          return;
        }

        // Lock scanning until reset
        scanningLockRef.current = true;

        // Extract token using utility (security: extractTokenFromQR never logs content)
        const result = extractTokenFromQR(decodedText);

        if (result.token) {
          // SECURITY: Pass token to callback but NEVER log it
          onScan(result.token);
        } else {
          // Release lock on error so user can scan again
          scanningLockRef.current = false;
          onError?.(result.error || 'Invalid QR code');
        }
      }, [onScan, onError]);

      useEffect(() => {
        // SSR safety check
        if (typeof window === 'undefined') return;

        // Mark initialization as started for cleanup handling
        initRef.current = true;

        // HIGH PRIORITY: HTTPS development check
        if (!isSecureContext()) {
          setState('unsupported');
          setErrorMessage('Camera requires HTTPS or localhost. Please access via https:// or localhost:3000');
          return;
        }

        // LOW PRIORITY: iOS version detection
        if (isUnsupportedIOS()) {
          setState('unsupported');
          setErrorMessage('Camera scanning requires iOS 14.3 or later. Please use manual order entry below.');
          return;
        }

        // MEDIUM PRIORITY: Rely on NotAllowedError/NotFoundError, not permissions.query
        const checkCameraAccess = async () => {
          try {
            const stream = await navigator.mediaDevices?.getUserMedia({
              video: { facingMode: { ideal: 'environment' } } // LOW PRIORITY: facingMode preference
            });
            stream?.getTracks().forEach(track => track.stop());
            initializeScanner();
          } catch (err) {
            const error = err as Error;

            // MEDIUM PRIORITY: Handle specific errors, don't query permissions API
            if (error.name === 'NotAllowedError' || error.name === 'PermissionDeniedError') {
              setState('permission-denied');
            } else if (error.name === 'NotFoundError' || error.name === 'DevicesNotFoundError') {
              setState('unsupported');
              setErrorMessage('No camera found on this device.');
            } else if (error.name === 'NotSupportedError' || error.name === 'SecurityError') {
              setState('unsupported');
              setErrorMessage('Camera access requires a secure connection (HTTPS) or localhost.');
            } else {
              setState('error');
              setErrorMessage(error.message || 'Failed to access camera');
              onError?.(error.message);
            }
          }
        };

        checkCameraAccess();

        function initializeScanner() {
          try {
            const qrboxDimensions = getQrBoxDimensions();

            scannerRef.current = new Html5QrcodeScanner(
              elementId,
              {
                fps: 2,
                qrbox: qrboxDimensions, // MEDIUM PRIORITY: Responsive sizing
                aspectRatio: 1.0,
                showTorchButtonIfSupported: true,
                showZoomSliderIfSupported: true,
              },
              false
            );

            scannerRef.current.render(
              handleScan,
              (error) => {
                // SECURITY: Never log error content if it contains scanned data
                if (error?.includes('No MultiFormat Readers')) return;
                if (error?.includes('No barcode found')) return;

                if (error?.includes('NotAllowedError')) {
                  setState('permission-denied');
                }
              }
            );

            setState('scanning');
          } catch (err) {
            const error = err as Error;
            setState('error');
            setErrorMessage(error.message || 'Failed to start camera');
            onError?.(error.message);
          }
        }

        // MEDIUM PRIORITY: Cleanup safeguards - runs even if init incomplete
        return () => {
          if (scannerRef.current) {
            scannerRef.current.clear().catch(() => {
              // Silent cleanup failure is acceptable
            });
          }
        };
      }, [handleScan, onError, getQrBoxDimensions]);

      // Permission denied state with browser-specific guidance
      if (state === 'permission-denied') {
        return (
          <div className="text-center p-6 bg-amber-50 rounded-xl border border-amber-200">
            <CameraOff className="w-12 h-12 text-amber-600 mx-auto mb-4" />
            <h3 className="font-semibold text-amber-900 mb-2">Camera Access Denied</h3>
            <p className="text-amber-700 text-sm mb-4">
              {getPermissionInstructions()}
            </p>
            <p className="text-amber-600 text-xs mb-4">
              Or use manual order entry below.
            </p>
            <Button onClick={() => window.location.reload()} variant="outline" className="bg-white">
              Try Again
            </Button>
          </div>
        );
      }

      // Unsupported browser state
      if (state === 'unsupported') {
        const isHttpsIssue = errorMessage.includes('HTTPS') || errorMessage.includes('localhost');

        return (
          <div className="text-center p-6 bg-gray-50 rounded-xl border border-gray-200">
            <CameraOff className="w-12 h-12 text-gray-500 mx-auto mb-4" />
            <h3 className="font-semibold text-gray-900 mb-2">Camera Not Available</h3>
            <p className="text-gray-600 text-sm mb-4">{errorMessage}</p>
            {isHttpsIssue && (
              <div className="bg-amber-50 border border-amber-200 rounded-lg p-4 text-left text-sm mb-4">
                <p className="font-medium text-amber-900 mb-2">To fix this:</p>
                <ol className="text-amber-800 space-y-1 list-decimal list-inside">
                  <li>Use <code className="bg-amber-100 px-1 rounded">localhost:3000</code> instead of IP address</li>
                  <li>Or set up HTTPS for your local network (e.g., using ngrok)</li>
                </ol>
                <p className="mt-3 text-amber-700">
                  Browsers require secure connections for camera access.
                </p>
              </div>
            )}
            <p className="text-gray-500 text-sm">Use manual order entry below.</p>
          </div>
        );
      }

      // Error state
      if (state === 'error') {
        return (
          <div className="text-center p-6 bg-red-50 rounded-xl border border-red-200">
            <AlertCircle className="w-12 h-12 text-red-600 mx-auto mb-4" />
            <h3 className="font-semibold text-red-900 mb-2">Scanner Error</h3>
            <p className="text-red-700 text-sm">{errorMessage}</p>
          </div>
        );
      }

      // Initializing or scanning state
      return (
        <div className="relative">
          {state === 'initializing' && (
            <div className="absolute inset-0 flex items-center justify-center bg-cream-50 z-10">
              <div className="text-center">
                <Loader2 className="w-8 h-8 text-forest-600 animate-spin mx-auto mb-2" />
                <p className="text-forest-600 text-sm">Initializing camera...</p>
              </div>
            </div>
          )}
          <div id={elementId} className="w-full" />
        </div>
      );
    }
    ```

    'use client';

    import dynamic from 'next/dynamic';
    import { Suspense } from 'react';
    import { Loader2 } from 'lucide-react';

    // Dynamic import with SSR disabled - REQUIRED for browser-only camera APIs
    const QRScanner = dynamic(
      () => import('./QRScanner').then(mod => mod.QRScanner),
      {
        ssr: false,
        loading: () => (
          <div className="flex items-center justify-center h-64 bg-cream-50 rounded-xl">
            <div className="text-center">
              <Loader2 className="w-8 h-8 text-forest-600 animate-spin mx-auto mb-2" />
              <p className="text-forest-600 text-sm">Loading scanner...</p>
            </div>
          </div>
        )
      }
    );

    interface ScannerContainerProps {
      onScan: (token: string) => void;
      onError?: (error: string) => void;
    }

    export function ScannerContainer({ onScan, onError }: ScannerContainerProps) {
      return (
        <Suspense fallback={
          <div className="flex items-center justify-center h-64 bg-cream-50 rounded-xl">
            <div className="text-center">
              <Loader2 className="w-8 h-8 text-forest-600 animate-spin mx-auto mb-2" />
              <p className="text-forest-600 text-sm">Loading scanner...</p>
            </div>
          </div>
        }>
          <QRScanner onScan={onScan} onError={onError} />
        </Suspense>
      );
    }
    ```

    This wrapper is CRITICAL for Next.js - it prevents SSR errors because:
    1. html5-qrcode accesses window/navigator on import
    2. Dynamic import with ssr: false ensures it only loads on client
    3. Suspense provides loading state during dynamic import
